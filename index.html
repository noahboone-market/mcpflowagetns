Quick Start
mcp.config.ts 
import { createRouter, defineAgents } from 'mcp-flow';

const agents = defineAgents({
  git:      { model: 'gpt-4',   tools: ['github'] },
  codegen:  { model: 'claude',  tools: ['react']  },
  security: { model: 'claude',  tools: ['lint']   },
});

export default createRouter({ agents });
